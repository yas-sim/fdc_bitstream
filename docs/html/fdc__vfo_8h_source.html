153 double phase_err_diff = phase_error - m_prev_phase_error;
154 double phase_correction = (phase_error * 0.02f /*- phase_err_diff * 0.002f*/) * m_current_gain; // PD control (no I element)
155 constexpr double phase_correction_limitter = 0.125f;
156 phase_correction = limit(phase_correction, -m_cell_size_ref * phase_correction_limitter,
157 m_cell_size_ref * phase_correction_limitter);
158 pulse_pos += phase_correction;
159 m_prev_phase_error = phase_error;
160
