91// ---------- Data separator

93// How data separator work:
94//
95// | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | <- Data window(W), Data cell boundary(|)
96// P P P P P <- Data pulses
97// 1 0 1 0 0 1 0 0 <- Data reading (only the data pulses within the data window will be considered)
98//
99// 01234567
100// | WWWW | <= In this case, bit cell size = 8, data window ofst = 2, data window size = 4
101//
102int mfm_codec::read_bit_ds(void) {
103 int bit_reading = 0;
104 double cell_center;
105
106 if (is_track_ready() == false) {
107 return -1;
108 }
109
110 do {
111 // check if the next bit is within the next data window
112 if (m_distance_to_next_pulse < m_bit_cell_size) {
113 // check pulse position
114 if (m_distance_to_next_pulse >= m_data_window_ofst &&
115 // regular pulse (within the data window)
116 m_distance_to_next_pulse < m_data_window_ofst + m_data_window_size) {
117 bit_reading = 1;
118 }
119 else {
120 // irregular pulse
121#ifdef DEBUG
122 std::cout << '?';
123#endif
124 }
125 // Adjust pulse phase (imitate PLL operation)
126 // Limit the PLL operation frequency and introduce fluctuation with the random generator (certain fluctuation is required to reproduce some copy protection)
127 if (m_fluctuation == false || m_rnd() % m_fluctuator_denominator >= m_fluctuator_numerator) {
128 cell_center = m_data_window_ofst + m_data_window_size / 2.f;
129 double error = m_distance_to_next_pulse - cell_center;
130
131 // Data pulse position adjustment == phase correction
132 m_distance_to_next_pulse -= error * 0.05f * m_pll_gain;
133
134 // Cell size adjustment == frequency correction
135 double new_cell_size = m_bit_cell_size + error * 0.01f * m_pll_gain;
136 // Limit the range of cell size
137 if (new_cell_size < m_bit_cell_size_ref * 0.8) new_cell_size = m_bit_cell_size_ref * 0.8;
138 if (new_cell_size > m_bit_cell_size_ref * 1.2) new_cell_size = m_bit_cell_size_ref * 1.2;
139 set_cell_size(new_cell_size);
140 }
141 size_t distance = m_track.distance_to_next_bit1();
142#if 0
143 // give timing fluctuation (intentionally - to immitate the spndle rotation fluctuation)
144 size_t rand_num = m_rnd() % 32;
145 if (rand_num == 0) {
146 distance++;
147 }
148 else if (rand_num == 1) {
149 distance--;
150 }
151#endif
152 m_distance_to_next_pulse += distance;
153 }
154 } while (m_distance_to_next_pulse < m_bit_cell_size);
155 // advance bit cell position
156 if (m_distance_to_next_pulse >= m_bit_cell_size) {
157 m_distance_to_next_pulse -= m_bit_cell_size;
158 }
159 return bit_reading;
160}

181bool mfm_codec::mfm_read_byte(uint8_t& data, bool& missing_clock, bool ignore_missing_clock, bool ignore_sync_field) {
182 size_t decode_count = 0;
183 missing_clock = false;
184
185 if (is_track_ready() == false) {
186 data = 0;
187 missing_clock = false;
188 return false;
189 }
190
191 do {
192 int bit_data = read_bit_ds();
193 if (m_track.is_wraparound()) {
194 m_track.clear_wraparound_flag();
195 m_wraparound = true;
196 }
197 m_bit_stream = (m_bit_stream << 1) | bit_data;
198
199 if (ignore_missing_clock == false) {
200 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_a1) { // Missing clock 0xA1 pattern
201 data = 0xa1;
202 missing_clock = true;
203 return true;
204 }
205 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_c2) { // Missing clock 0xC2 pattern
206 data = 0xc2;
207 missing_clock = true;
208 return true;
209 }
210 }
211 if (ignore_sync_field == false) {
212 if (m_bit_stream == m_pattern_00) {
213 decode_count &= ~0b01u; // C/D synchronize
214 set_gain(10.f);
215 }
216 else {
217 set_gain(1.f);
218 }
219 }
220 } while (++decode_count < 16);
221
222 // Extract only 'D' bits (exclude 'C' bits). MFM data is 'CDCDCDCD..CD'.
223 uint8_t read_data = 0;
224 for (uint16_t bit_pos = 0x4000; bit_pos > 0; bit_pos >>= 2) {
225 read_data = (read_data << 1) | (m_bit_stream & bit_pos ? 1 : 0);
226 }
227 data = read_data;
228 missing_clock = false;
229 return true;
230}

233// mode: false=normal, true=special(write track, etc. cares 'F5' and 'F6')
234uint16_t mfm_codec::mfm_encoder(uint8_t data, bool mode) {
235 // Data swap for special code
236 uint8_t write_data = data;
237 if (mode == true) { // special coding mode (F5 and F6 will be converted to A1* C2* with missing-clock pattern respectively)
238 switch (data) {
239 case 0xf5:
240 write_data = 0xa1;
241 break;
242 case 0xf6:
243 write_data = 0xc2;
244 break;
245 }
246 }
247 // MFM encoding
248 uint16_t bit_pattern = 0;
249 uint16_t mfm_bit_pattern = 0;
250 int current_bit;
251 int clock_bit;
252 for (size_t bit_pos = 0x80; bit_pos != 0; bit_pos >>= 1) {
253 current_bit = (write_data & bit_pos) ? 1 : 0;
254 clock_bit = (m_prev_write_bit == 0 && current_bit == 0) ? 1 : 0;
255 mfm_bit_pattern = (mfm_bit_pattern << 1) | clock_bit; // clock
256 mfm_bit_pattern = (mfm_bit_pattern << 1) | current_bit; // data
257 m_prev_write_bit = current_bit;
258 }
259 // Missing clock operation
260 if (mode == true) {
261 switch (data) {
262 case 0xf5:
263 mfm_bit_pattern &= ~0b100000u; // remove a clock bit
264 break;
265 case 0xf6:
266 mfm_bit_pattern &= ~0b10000000u; // remove a clock bit
267 break;
268 }
269 }
270 return mfm_bit_pattern;
271}

273// mode: false=normal, true=special(write track, etc)
274// write_gate: true=perform actual write false=dummy write (no actual write will be perfored)
275void mfm_codec::mfm_write_byte(uint8_t data, bool mode, bool write_gate) {
276 if (is_track_ready() == false) {
277 return;
278 }
279 uint16_t bit_pattern = mfm_encoder(data, mode);
280 for (uint16_t bit_pos = 0x8000; bit_pos != 0; bit_pos >>= 1) {
281 int bit = bit_pattern & bit_pos ? 1 : 0;
282 if (write_gate == true) {
283 for (size_t i = 0; i < m_bit_width; i++) {
284 if (m_bit_width / 2 == i) m_track.write_stream(bit); // write data pulse at the center of the bit cell
285 else m_track.write_stream(0);
286 }
287 }
288 else {
289 for (size_t i = 0; i < m_bit_width; i++) {
290 m_track.advance_stream_pos(); // advance stream pointer without actual data write (dummy write)
291 }
292 }
293 }
